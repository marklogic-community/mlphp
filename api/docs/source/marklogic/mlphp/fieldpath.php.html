namespace MarkLogic\MLPHP;
 
use JsonSerializable;
 
/**
 * Represents a database field path.
 *
 * @package MLPHP
 * @author Mike Wooldridge <mike.wooldridge@marklogic.com>
 */
class FieldPath implements JsonSerializable
{
    public $properties = array(); // @var array
 
    /**
     * Create a FieldPath object.
     *
     * @param array $properties Associative array of config properties.
     */
    public function __construct($properties)
    {
        $this->properties = array_merge(array(
            'path' => '',
            'weight' => 1
        ), $properties);
    }
 
    public function jsonSerialize() {
        return $this->properties;
    }
 
}
